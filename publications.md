<script>
async function injectOpenAlexCitations() {
  const cards = document.querySelectorAll(".pub-card");

  for (const card of cards) {
    const badge = card.querySelector(".altmetric-embed");
    if (!badge || !badge.dataset.doi) continue;

    const line = document.createElement("div");
    line.className = "citation-line";
    line.textContent = "Citations (OpenAlex): loading…";
    card.querySelector(".pub-content").appendChild(line);

    try {
      const r = await fetch(
        `https://api.openalex.org/works/doi:${encodeURIComponent(badge.dataset.doi)}`
      );
      const j = await r.json();
      line.textContent = `Citations (OpenAlex): ${j.cited_by_count ?? "N/A"}`;
    } catch {
      line.textContent = "Citations (OpenAlex): N/A";
    }
  }
}

window.addEventListener("load", injectOpenAlexCitations);
</script>
